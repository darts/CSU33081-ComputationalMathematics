function [L, U] = LUdecompGauss (A)
    [x, y] = size(A)
    if x <= 0 || x ~= y
        disp("Matrix Is Not Square!");
        return
    end

    L = eye(x, x);
    U = A;

    for i = 1:x
        for j = i + 1:x
            c = U(j, i) / U(i, i);
            for k = 1:x
                U(j, k) = U(j, k) - (c * U(i, k));
            end
            L(j, i) = c;
        end
    end
end
