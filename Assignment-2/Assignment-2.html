function [L, U] = LUdecompGauss (A)
    [x, y] = size(A)
    if x <= 0 || x ~= y
        disp("Matrix Is Not Square!");
        return
    end

    L = eye(x, x);
    U = A;

    for i = 1:x
        for j = i + 1:x
            c = U(j, i) / U(i, i);
            for k = 1:x
                U(j, k) = U(j, k) - (c * U(i, k));
            end
            L(j, i) = c;
        end
    end
end

function [a1, a0] = LinearReg(x,y)
    nx=length(x);
    ny=length(y);
    if nx ~= ny
        disp('ERROR: The number of elements in x must be the same as in y.');
        a1 = 'Error';
        a0 = 'Error';
    else
        Sx = sum(x);
        Sy = sum(y);
        Sxy = sum(x.*y);
        Sxx = sum(x.^2);
        a1 = (nx*Sxy-Sx*Sy)/(nx*Sxx-Sx^2);
        a0 = (Sxx*Sy-Sxy*Sx)/(nx*Sxx-Sx^2);
    end
end
